/* A program to perform Euclid's
   Algorithm to compute gcd.  */

int a;
int b;

int gcd(int u, int v, int other) {
  if (v == 0) {
    return u;
  } else {
    /* u-u/v*v == u mod v */
    return gcd(v, u - u / v *v, u - u / v *v);
  }
}

  int x;
