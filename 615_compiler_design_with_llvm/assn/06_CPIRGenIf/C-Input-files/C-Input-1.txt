/* A program to perform Euclid's
   Algorithm to compute gcd.  */

int gcd(int u, int v) {
  int r;
  if (v == 0) {
    r = u;
  } else {
    /* u-u/v*v == u mod v */
    r = gcd(v,u-u/v*v);
  }
  return r;
}

void main(void) {
  int x; int y;
  x = input();
  y = input();
  output(gcd(x,y));
}
