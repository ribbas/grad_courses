/* A program to perform Euclid's
   Algorithm to compute gcd.  */

int gcd(int u, int v) {
  if (v == 0) {
    //u = v + 10;
    return u;
  } else {
    /* u-u/v*v == u mod v */
    //u = v + 20;
    return gcd(v, u-u/v*v);
  }
  //return u;
}

void main(void) {
  int x; int y;
  x = input();
  y = input();
  output(gcd(x,y));
  return;
}
